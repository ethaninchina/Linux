-  格式化分区
```
# mkfs.ext4 /dev/vdb1
```

-  挂载分区
```
# mkdir /data
# mount /dev/vdb1 /data/
```

-  挂载写入配置
```
# blkid /dev/vdb1
/dev/vdb1: UUID="520ff46c-753a-4063-a6f2-5d7ff0439f33" TYPE="ext4"


# vi /etc/fstab
-----------------------------------------------------------------
UUID=520ff46c-753a-4063-a6f2-5d7ff0439f33 /data			  ext4    defaults        0 0
-----------------------------------------------------------------
```
